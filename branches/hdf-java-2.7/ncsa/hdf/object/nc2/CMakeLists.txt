cmake_minimum_required (VERSION 2.8.3)
PROJECT (HDFJAVA_NCSA_HDF_OBJECT_NC2 Java)

SET (CMAKE_VERBOSE_MAKEFILE 1)

INCLUDE_DIRECTORIES (
    ${HDFJAVA_NCSA_HDF_OBJECT_NC2_SOURCE_DIR}
    ${HDFJAVA_NCSA_HDF_OBJECT_NC2_BINARY_DIR}
)

SET (HDFJAVA_NCSA_HDF_OBJECT_NC2_SRCS
    NC2Dataset.java
    NC2Datatype.java
    NC2File.java
    NC2Group.java
)

INCLUDE_DIRECTORIES (${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/${HDFJAVA_NCSA_NC2_LIB_CORENAME}.dir/ncsa/hdf/object/nc2)
SET (CMAKE_JAVA_INCLUDE_PATH ${CMAKE_JAVA_INCLUDE_PATH} ${${HDFJAVA_NCSA_OBJECT_LIB_TARGET}_JAR_FILE})

ADD_JAR (${HDFJAVA_NCSA_NC2_LIB_TARGET} "." ${HDFJAVA_NCSA_HDF_OBJECT_NC2_SRCS})
INSTALL_JAR (${HDFJAVA_NCSA_NC2_LIB_TARGET} ${HJAVA_INSTALL_LIB_DIR} libraries)
SET_GLOBAL_VARIABLE (HDFJAVA_JARS_TO_EXPORT "${HDFJAVA_JARS_TO_EXPORT};${${HDFJAVA_NCSA_NC2_LIB_TARGET}_JAR_FILE}")

ADD_DEPENDENCIES (${HDFJAVA_NCSA_NC2_LIB_TARGET} ${HDFJAVA_NCSA_OBJECT_LIB_TARGET})
